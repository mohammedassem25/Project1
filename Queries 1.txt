/* Query 1 */ The Top Genre
SELECT G.GenreId,
       G.Name Genre_Name,
       COUNT(C.CUSTOMERID) AS Customers_Num,
       SUM(IL.UnitPrice * IL.Quantity) AS Total_Cost
FROM Customer C
JOIN Invoice I ON I.CustomerId = C.CustomerId
JOIN InvoiceLine IL ON IL.InvoiceId = I.InvoiceId
JOIN Track T ON T.TrackId = IL.TrackId
JOIN Genre G ON G.GenreId = T.GenreId
GROUP BY 1
ORDER BY 4 DESC, 3 DESC
LIMIT 10;

/* Query 2 */ The Best Album
SELECT A.ALBUMID,
       G.NAME GENRE_NAME,
       A.TITLE AS ALBUM_NAME,
       COUNT( I.CustomerId ) AS TOTAL_Customers
FROM Invoice I
JOIN InvoiceLine IL ON IL.InvoiceId = I.InvoiceId
JOIN Track T ON T.TrackId = IL.TrackId
JOIN Genre G ON G.GenreId = T.GenreId
JOIN Album A ON A.AlbumId = T.AlbumId
GROUP BY 1,3
ORDER BY 4 DESC
LIMIT 10;

/* Query 3 */ Top 10 Preferable Artists
WITH BEST_ALBUM AS
  (SELECT A.ALBUMID,
          G.NAME GENRE_NAME,
          A.TITLE AS ALBUM_NAME,
          AR.ARTISTID,
          AR.NAME AS ARTIST_NAME,
          COUNT(C.CUSTOMERID) AS Num_CUSTOMERS,
	  SUM(IL.UnitPrice * IL.Quantity) AS TOTAL_DOLLARS
   FROM Customer C
   JOIN Invoice I ON I.CustomerId = C.CustomerId
   JOIN InvoiceLine IL ON IL.InvoiceId = I.InvoiceId
   JOIN Track T ON T.TrackId = IL.TrackId
   JOIN Genre G ON G.GenreId = T.GenreId
   JOIN Album A ON A.AlbumId = T.AlbumId
   JOIN Artist AR ON AR.ArtistId = A.ArtistId
   GROUP BY 1,3
   ORDER BY 7 DESC)
SELECT A.ALBUMID,
       AR.ARTISTID,
       AR.NAME AS ARTIST_NAME,
       BEST_ALBUM.TOTAL_DOLLARS
FROM BEST_ALBUM
JOIN Album A ON A.AlbumId = BEST_ALBUM.AlbumId
JOIN Artist AR ON AR.ArtistId = A.ArtistId
LIMIT 10;

/* Query 4 */ Top Rated Countries
SELECT C.Country AS COUNTRY,
       SUM(IL.UnitPrice * IL.Quantity) AS Total_Dollars,
       CASE
           WHEN SUM(IL.UnitPrice * IL.Quantity) > 160 THEN 'TOP_COUNTRY'
           WHEN SUM(IL.UnitPrice * IL.Quantity) BETWEEN 80 AND 160 THEN 'MED_COUNTRY'
           ELSE 'LOW_COUNTRY'
       END AS COUNTRIES_RATE
FROM Customer C
JOIN Invoice I ON I.CustomerId = C.CustomerId
JOIN InvoiceLine IL ON IL.InvoiceId = I.InvoiceId
JOIN Track T ON T.TrackId = IL.TrackId
GROUP BY 1
ORDER BY 2 DESC
LIMIT 10;
